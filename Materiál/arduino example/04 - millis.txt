#include <LiquidCrystal.h>

LiquidCrystal lcd(8, 7, 6, 5, 4, 3);

long cas;
int cislo = 0;

void setup()
{
  Serial.begin(9600);
  lcd.begin(16, 2);
  cas = millis() + 1000;
}

void loop()
{
  if ( millis() > cas)
  {
    cislo++;
    lcd.clear();
    lcd.print(cislo);
    cas = millis() + 1000;
  }

  if ( Serial.available() )
  {
    int cisielko = Serial.parseInt();
    //Serial.print(cisielko);
    if (cisielko != 0)
    {
      cislo = cisielko - 1;
    }
    cas = 0;
  }
}
